Here's a Python script that reads a CSV and generates a visual summary:

import pandas as pd
import matplotlib.pyplot as plt

def analyze_csv(filepath: str) -> None:
    """Read CSV and generate summary charts."""
    df = pd.read_csv(filepath)

    # Print summary statistics
    print(df.describe())

    # Generate charts for numeric columns
    numeric_cols = df.select_dtypes(include=["number"]).columns
    fig, axes = plt.subplots(
        1, len(numeric_cols),
        figsize=(5 * len(numeric_cols), 4)
    )

    for i, col in enumerate(numeric_cols):
        ax = axes[i] if len(numeric_cols) > 1 else axes
        df[col].hist(ax=ax, bins=20, color="#3b82f6")
        ax.set_title(col)

    plt.tight_layout()
    plt.savefig("summary.png", dpi=150)
    print("Chart saved to summary.png")

if __name__ == "__main__":
    analyze_csv("data.csv")
